Desi hesabi en*boy*yükseklik/3000 formulü ile bulunmaktadir. Buna göre verilen en, boy, ve yükseklik bilgilerini kullanarak desi hesabi yapip sonucunu dönen bir fonksiyon yaziniz.
Verilen bir oyuncu için toplam kazandigi oyun sayisini dönen bir fonksiyon yaziniz. Eger oyuncunun kazandigi hiç oyun yoksa sonuç NULL dönmemelidir.
Verilen boy (metre cinsinden) ve kilo (kg cinsinden) degerlerine göre Vücut Kitle Indikesi (Body Mass Index, BMI) degerini geri dönen bir fonksiyon yaziniz . BMI degeri, KILO/ (BOY * BOY) seklinde hesaplanmaktadir. Sonuç degeri noktadan sonra iki basamak olacak sekilde yuvarlanarak döndürülmelidir
Verilen yüzlük (100) not degerini dörtlük (4) not degerine çevirecek fonksiyonu yaziniz . Çevrim için formül: DörtÜzerindenNot = -0.283 + 0.043 * YüzÜzerindenNot

Example 1.

/*	Desi hesabi en*boy*yükseklik/3000 formulü ile bulunmaktadir. Buna göre verilen en, boy, ve yükseklik bilgilerini kullanarak	desi hesabi yapip sonucunu dönen bir fonksiyon yaziniz.*/

CREATE FUNCTION fncDesi(@en FLOAT, @boy FLOAT, @yukseklik FLOAT) RETURNS FLOAT AS 
BEGIN 
	RETURN @en * @boy * @yukseklik / 3000
END
GO

Example 2.

/*Verilen bir oyuncu için toplam kazandigi oyun sayisini dönen bir fonksiyon yaziniz. Eger oyuncunun kazandigi hiç oyun yoksa sonuç NULL dönmemelidir.*/
CREATE FUNCTION fncKazandigiOyunSayisi(@OyuncuID INT) RETURNS INT AS 
BEGIN 
	RETURN (SELECT ISNULL(COUNT(*), 0)FROM tblOyun O WHERE O.Kazanan = @OyuncuID)
-- Sonucun NULL cikmamasi için ISNULL fonksiyonu kullanildi

END 
GO

Example 3.

/*	Verilen boy (metre cinsinden) ve kilo (kg cinsinden) degerlerine göre Vücut Kitle Indikesi (Body Mass Index, BMI) degerini geri dönen bir fonksiyon yaziniz .	BMI degeri, KILO/ (BOY * BOY) seklinde hesaplanmaktadir.	Sonuç degeri noktadan sonra iki basamak olacak sekilde yuvarlanarak döndürülmelidir*/

CREATE FUNCTION fncBMI(@boy FLOAT, @kilo FLOAT) RETURNS FLOAT AS 
BEGIN
	RETURN ROUND(@kilo / (@boy * @boy), 2)
END
GO

/*Verilen yüzlük (100) not degerini dörtlük (4) not degerine çevirecek fonksiyonu yaziniz . Çevrim için formül:	DörtÜzerindenNot = -0.283 + 0.043 * YüzÜzerindenNot*/

	RETURN -0.283 + 0.0429 * @yuzluk_not

/*	Her bir tedarikçinin (supplier) ismini ve kendisinden kaç farkli  ürün alindigini gösteren bir view yaziniz .	Ornegin Tokyo Traders firmasindan Mishi Kobe Niku, Ikura, ve Longlife Tofu ürünleri aliniyorsa bu view calistirildiginda Tokyo Traders için 3 yazacaktir.*/
CREATE VIEW vwSupplierProductAmount AS 
			SELECT	S.CompanyName,
							COUNT(P.ProductID)
			FROM Products P
			INNER JOIN Suppliers S ON S.SupplierID = P.SupplierID
			GROUP BY S.CompanyName
GO

/*
TÜM oyuncular için asagidaki sekilde cikti verebilecek bir VIEW yaziniz

	O.ogrno AS OgrNo,

/*	Sistemde her bir satis elemanin sorumlu oldugu bölgeler EmployeeTerritories tablosunda tanimlanmistir. Bir bölgeden yalnizca bir eleman sorumlu olmali ve bir elemanin en çok sorumlu oldugu bölge sayisi da 10 olmalidir.	Buna göre istenen bir satis elemani için bölge atamasi yapabilen bir SP yaziniz. Ayrica bahsedilen gerekli kontrolleri de	bir trigger ile gerceklestiriniz*/
CREATE PROCEDURE spAddEmpTerritory(@EmployeeID int, @TerritoryID nvarchar(20)) AS 
	INSERT INTO EmployeeTerritories VALUES(@EmployeeID , @TerritoryID)
GO 
CREATE TRIGGER trgCheckEmpTerritory ON EmployeeTerritories AFTER INSERT AS 
	DECLARE @empid INT, @terid nvarchar(20), @c INT 
	SELECT @empid = EmployeeID, @terid = TerritoryID FROM inserted
	SELECT @c = COUNT(TerritoryID) FROM EmployeeTerritories WHERE EmployeeID = @empid
	IF (@c - 1) = 10
		BEGIN 
				ROLLBACK 
				RAISERROR('THIS EMPLOYEE ALREADY HAS 10 TERRITORIES...', 16, 1)
		END 
	IF EXISTS (SELECT *FROM EmployeeTerritories WHERE TerritoryID = @terid)
		BEGIN 
				ROLLBACK
				RAISERROR('THIS(%s) TERRITORIY BELONGS ANOTHER EMPLOYEE...', 16, 1, @terid)
		END
GO

/*	Sipariþ ayrýntýlarý (Order Details) tablosuna istenen bir sipariþe ait bir ürün için satýr ekleyebilecek bir SP yazýnýz.	Ekleme yapýlýrken þu kýsýtlarý göz önünde bulundurun		* Eðer stok yetersiz ise ekleme yapýlamaz ve uyarý verilir		* Ekleme yapýlabiliyorsa, ürüne ait stok miktarýnýn da düþürülmesi gerekir		* Bir trigger vasýtasýyla stok miktarý, yeniden sipariþ seviyesine (ReorderLevel) düþen ürün için		  “mal alým sipariþi verin” þeklinde bir mesaj da gösterilmelidir*/

CREATE PROCEDURE spAddOrderDetails(@OrderID INT, @ProductID INT, @UnitPrice MONEY, @Quantity SMALLINT, @Discount REAL)AS 
			DECLARE @UnitsInStock SMALLINT = (SELECT UnitsInStock FROM Products WHERE ProductID = @ProductID)
			IF @UnitsInStock >= @Quantity
					BEGIN 
						INSERT INTO [Order Details] VALUES(@OrderID, @ProductID, @UnitPrice, @Quantity, @Discount)
						UPDATE ProductsSET UnitsInStock -= @Quantity WHERE ProductID = @ProductID
					END 
			ELSE 
					RAISERROR('INSUFFICENT STOCK...', 16, 1)
GO

CREATE TRIGGER trgCheckProductStock ON Products AFTER UPDATE AS 
		DECLARE @ReorderLevel SMALLINT, @UnitsInStock SMALLINT 
		SELECT @ReorderLevel = ReorderLevel, @UnitsInStock = UnitsInStock FROM inserted
		IF @UnitsInStock <= @ReorderLevel
			RAISERROR('LOW STOCK LEVEL... PLACE A PURCHASE ORDER...', 10, 1)

GO
